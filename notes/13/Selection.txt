\begin{definition} Zero-Divisors\\ A zero-divisor is a nonzero element a of a commutative ring $R$ such that there is a nonzero element $b\in R$ with $ab=0$.
\end{definition}
\begin{definition}Integral Domain\\
An integral domain is a commutative ring with unity and no zero-divisors.\end{definition}
\textbf{EXAMPLE 1}\\
The ring of integers is an integral domain.\\
\textbf{EXAMPLE 2}\\ The ring of Gaussian integers $\mathbb{Z}[i] = (a + bi \,\vert\, a, b\in\mathbb{Z})$ is an integral domain.\\
\textbf{EXAMPLE 3}\\ The ring $\mathbb{Z}[x]$ of polynomials with integer coefficients is an integral domain.\\
\textbf{EXAMPLE 4}\\ The ring $\mathbb{Z}[\sqrt{2}] = {a + b\sqrt{2} \,\vert\, a,b \in z}$ is an integral domain.\\
\textbf{EXAMPLE 5}\\ The ring, $Z_p$, of integers modulo a prime $p$ is an integral domain.\\
\textbf{EXAMPLE 6}\\ The ring $\mathbb{Z}_n$ of integers modulo $n$ is not an integral domain when $n$ is not prime.\\
\textbf{EXAMPLE 7}\\ The ring $M_2(\mathbb{Z})$ of $2\times 2$ matrices over the integers is not an integral domain.\\

\begin{definition} Field\\
A field is a commutative ring with unity in which every nonzero element is a unit.\\\end{definition}
\begin{theorem}A finite integral domain is a field.\end{theorem}
\begin{theorem}For every prime $p$, $\mathbb{Z}_p$, The ring of integers modulo $p$ is a field.\\\end{theorem}
\begin{definition} Characteristic of a Ring\\
The characteristic of a ring $R$ is the least positive integer $n$ such that nnx I 0 for all $x$ in $R$. If no such integer exists, we say that $R$ has characteristic 0. The characteristic of $R$ is denoted by $char(R)$.\\\end{definition}
\begin{theorem}Let $R$ be a ring with unity 1. If 1 has infinite order under addition, then the characteristic of $R$ is 0. If 1 has order $n$ under addition, then the characteristic of $R$ is $n$.\\\end{theorem}
\begin{theorem}The characteristic of an integral domain is 0 or prime.\\\end{theorem}
